129  /**
130  Get free blocks from the bitmap and set them to USED (or 1) in the bitmap.
131  */
132  private int[] get_free_blocks(int size) throws DiskException
133  {
134  bitmap_mutex.Lock();
135  int[] assembled = new int[size]; //make the array for the return
136  int target = 0; ///< the index for the integer in bitmap
137  int position = 0; ///< the position of the bit corresponding to the block to be written to
138  int filled = 0; ///< the number of free blocks we found (soon to be USED blocks.)
139  int temp;
140  for(int i=0;i<DATA_TOTAL;i++){
141  target = (i/BITMAP_SIZE); //this
142  position = (i%BITMAP_SIZE); //this reduces i to a number between 0-31, so it's position is relative to the int in bitmap[target]
143  temp = bitmap[target]; //copy the bitmap int
144  temp = temp >> position; //isolate the required bit
145  if(temp % 2 ==0){
146  //last bit is free
147  assembled[filled] = i;
148  ++filled;
149  }
150 
151  if(filled == size){
152  //we filled all slots.
153  break;
154  }
155 
156  }
157  if(filled < size){
158  throw new DiskException("Not Enough Free Blocks. Try deleting something first.");
159  }
160 
161  fill_blocks(assembled);
162  bitmap_mutex.UnLock();
163  return assembled;
